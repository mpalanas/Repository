from sikuli import*
import unittest
import json

class PasswordReset(unittest.TestCase):

    def setUp(self):

        wait(2)
        configPath = os.path.join(os.path.dirname(os.getcwd()), "SikuliScripts\Repository\config.json")
        config = json.loads(open(configPath).read())
        app = App(config["application"])
        openApp(app)
        wait(1)
        type("l", KeyModifier.CTRL)
        type(config["url"] + Key.ENTER)
        if exists():
            click(Pattern().targetOffset(104,3))
            click()
        wait(2)
        wait(2)

    def tearDown(self):
        click(Pattern().targetOffset(78,-3))
        click()
        wait(3)
        configPath = os.path.join(os.path.dirname(os.getcwd()), "SikuliScripts\Repository\config.json")
        config = json.loads(open(configPath).read())
        appTaskName = config["appTaskName"]
        os.system("taskkill /f /im " +appTaskName)

    def test_changePassword(self):
        configPath = os.path.join(os.path.dirname(os.getcwd()), "SikuliScripts\Repository\config.json")
        config = json.loads(open(configPath).read())
        click()
        click(Pattern().similar(0.62).targetOffset(-100,15))
        type(config["username"])
        type(Key.TAB)
        type(config["password"])
        click()
        wait(3)
        click(Pattern().targetOffset(79,-1))
        click()
        click(Pattern().targetOffset(-47,8))
        type(config["password"])
        click(Pattern().targetOffset(-38,0))
        type(config["password2"])
        click()
        wait(3)
        if exists():
            assert True
        else:
            assert False

    def test_loginNewPassword(self):
        configPath = os.path.join(os.path.dirname(os.getcwd()), "SikuliScripts\Repository\config.json")
        config = json.loads(open(configPath).read())
        click()
        click(Pattern().similar(0.62).targetOffset(-100,15))
        type(config["username"])
        type(Key.TAB)
        type(config["password2"])
        click()
        wait(3)
        if exists(Pattern().targetOffset(79,-1)):
            assert True
        else:
            assert False
