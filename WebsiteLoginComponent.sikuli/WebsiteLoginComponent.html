from sikuli import*
import unittest
import json

class WebsiteLoginComponent(unittest.TestCase):

    def setUp(self):
        configPath = os.path.join(os.path.dirname(os.getcwd()), "SikuliScripts\Repository\config.json")
        config = json.loads(open(configPath).read())
        app = App(config["application"])
        openApp(app)
        wait(1)
        type("l", KeyModifier.CTRL)
        type(config["url"] + Key.ENTER)
        wait(2)

    def tearDown(self):
        click(Pattern().targetOffset(104,3))
        click()
        wait(3)
        click(Pattern().targetOffset(26,0))

    def test_CreateAccount(self):
        configPath = os.path.join(os.path.dirname(os.getcwd()), "SikuliScripts\Repository\config.json")
        config = json.loads(open(configPath).read())
        click(Pattern().targetOffset(26,-4))
        click(Pattern().targetOffset(-133,12))
        type("FirstName")
        click(Pattern().targetOffset(-144,1))
        type("LastName")
        click()
        type(config["username"])
        click()
        type(config["password"])
        click()
        wait(4)

        if exists():
            assert True
        else:
            assert False
