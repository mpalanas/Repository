from sikuli import*
import unittest
import json

class FindUsBar(unittest.TestCase):
    def setUp(self):
        configPath = os.path.join(os.path.dirname(os.getcwd()), "SikuliScripts\Repository\config.json")
        config = json.loads(open(configPath).read())
        app = App(config["application"])
        openApp(app)
        wait(1)
        type("l", KeyModifier.CTRL)
        type(config["url"] + Key.ENTER)
        wait(2)
        find()
        wheel(, WHEEL_DOWN, 3)

    def tearDown(self):

        click(Pattern().targetOffset(24,0))

        wait(2)


    def test_FindABranch(self):
        click(Pattern().targetOffset(-86,38))
        if exists(Pattern().exact()):
            assert True
        else:
            assert False
    def test_OurTeam(self):
        click(Pattern().targetOffset(-8,37))
        if exists(Pattern().exact()):
            assert True
        else:
            assert False

    def test_FindPersonsName(self):
        click(Pattern().targetOffset(-97,6))
        click()
        type("bob voss")
        wait(2)
        if exists(Pattern().exact()):
            assert True
        else:
            assert False
